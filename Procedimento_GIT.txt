//Criar um novo repositório
git init


//configurar um novo repositório
git config --global user.name "Seu Nome"
git config --global user.email "seuemail@exemplo.com"


//adicionar modificações ao repositório local
git add .

//salvar modificações no repositório local
git commit -m "Descrição de modificações"


//vincular repositório local com o github
git remote add origin https://github.com/dev-filipealves/AulasBackend.git 

//enviar alterações para o GitHUB [Substituir master pela sua branch]
git push -u origin master

 
//em caso de erros, enviar a força as alterações
git push -f https://USUARIO:TOKEN@github.com/dev-filipealves/AulasBackend.git [BRANCH]
